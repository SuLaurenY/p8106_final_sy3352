load("severity.RData")
set.seed(3352)
my_dat <- dat[sample(nrow(dat), 1000), ]
my_dat <- my_dat |> select(-id)

my_dat$severity <- factor(my_dat$severity, levels = c("1", "0"),
                          labels = c("severe", "not_severe"))
my_dat$gender <- factor(my_dat$gender, levels = c(0, 1),
                        labels = c("Female", "Male"))
my_dat$race <- factor(my_dat$race, levels = c(1, 2, 3, 4),
                      labels = c("White", "Asian", "Black", "Hispanic"))
my_dat$smoking <- factor(my_dat$smoking, levels = c(0, 1, 2),
                         labels = c("Never", "Former", "Current"))
my_dat$hypertension <- factor(my_dat$hypertension, levels = c(0, 1),
                              labels = c("No", "Yes"))
my_dat$diabetes <- factor(my_dat$diabetes, levels = c(0, 1), 
                          labels = c("No", "Yes"))
my_dat$vaccine <- factor(my_dat$vaccine, levels = c(0, 1),
                         labels = c("Unvaccinated", "Vaccinated"))

set.seed(3352)
data_split <- initial_split(my_dat, prop = 0.8, strata = severity)
